#alias 64: TYPE_AMYROSE


event ObjectMain
	ArrayPos0 = -1
	foreach (GROUP_PLAYERS, currentPlayer, ACTIVE_ENTITIES)
		if object[currentPlayer].propertyValue == PLAYER_SONIC
			ArrayPos0 = currentPlayer
		end if
	next

	// This should never happen as we should always able to find
	// a player that corresponds to Sonic. Infact Amy will not
	// be initialised if she does not find Sonic.
	// this can happen if you add Knuckles or Metal for example
	if ArrayPos0 < 0
		ArrayPos0 = 0
	end if

	Object.Priority=1
	if object[ArrayPos0].XPos<Object.XPos
		Object.Direction=FACING_LEFT
	else
		Object.Direction=FACING_RIGHT
	end if
	switch Object.State
	case 0
		temp0=Object.Value6
		temp0/=10
		temp0+=11
		Object.Frame=temp0
		Object.Value6++
		if Object.Value6>139
			Object.Value6=0
		end if
		
    	BoxCollisionTest(C_TOUCH, object.entityPos, -96,-360,96,360, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		if CheckResult==1
			Object.Value0=0
			Object.State=1
			Object.Frame=4
			Object.Value6=0
		end if
		break
	case 1
		Object.Value6++
		switch Object.Frame
		case 4
		case 6
			if Object.Value6==4
				Object.Value6=0
				Object.Frame++
			end if
			break
		case 5
		case 7
			if Object.Value6==3
				Object.Value6=0
				Object.Frame++
			end if
			break

		end switch
		if Object.Frame>7
			Object.Frame=4
		end if
		if Object.Direction==FACING_RIGHT
			if Object.Value0<131072
				Object.Value0+=4096
			end if

			BoxCollisionTest(C_TOUCH, object.entityPos, -32,-360,4,360, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if CheckResult==1
				if object[ArrayPos0].Gravity==0
					if object[ArrayPos0].Direction==Object.Direction
						BoxCollisionTest(C_TOUCH, object.entityPos, -32,-20,4,20, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
						if CheckResult==1
							CheckEqual(object[ArrayPos0].Animation,ANI_SPINDASH)
							temp0=CheckResult
							CheckEqual(Object[2].Type,TypeName[BlueShield])
							temp0|=CheckResult
							if temp0==1
								Object.State=8
								Object.Frame=9
								Object.Value1=-262144
								Object.Value5=1
								Object.Value0=-65536
							else
								Object.Value4=1
								Object.State=7
								Object.Frame=8
								Object.Value6=0
								object[ArrayPos0].State=PlayerObject_Hugged
								//PlayStageSfx(3,0) // TODO
								if Stage.DebugMode==0
									//SetAchievement(1,100) // TODO
								end if
							end if
						else
							Object.State=6
							Object.Frame=0
							Object.Value6=0
							Object.Value0=0
						end if
					else
						Object.State=6
						Object.Frame=0
						Object.Value6=0
						Object.Value0=0
					end if
				end if
			end if
			if Object.XPos<Object.Value3
				Object.XPos+=Object.Value0
			else
				Object.Value0=0
				Object.XPos=Object.Value3
				if Object.Value4==1
					Object.State=4
				else
					Object.State=2
					Object.Frame=0
					Object.Value6=0
				end if
			end if
		else
			if Object.Value0>-131072
				Object.Value0-=4096
			end if

			BoxCollisionTest(C_TOUCH, object.entityPos, -4,-360,32,360, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
			if CheckResult==1
				if object[ArrayPos0].Gravity==0
					if object[ArrayPos0].Direction==Object.Direction
						BoxCollisionTest(C_TOUCH, object.entityPos, -32,-20,4,20, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
						if CheckResult==1
							CheckEqual(object[ArrayPos0].Animation,ANI_SPINDASH)
							temp0=CheckResult
							CheckEqual(Object[2].Type,TypeName[BlueShield])
							temp0|=CheckResult
							if temp0==1
								Object.State=8
								Object.Frame=9
								Object.Value1=-262144
								Object.Value5=1
								Object.Value0=65536
							else
								Object.Value4=1
								Object.State=7
								Object.Frame=8
								Object.Value6=0
								object[ArrayPos0].State=PlayerObject_Hugged
								//PlayStageSfx(3,0) // TODO
								if Stage.DebugMode==0
									//SetAchievement(1,100) // TODO
								end if
							end if
						else
							Object.State=6
							Object.Frame=0
							Object.Value6=0
							Object.Value0=0
						end if
					else
						Object.State=6
						Object.Frame=0
						Object.Value6=0
						Object.Value0=0
					end if
				end if
			end if
			if Object.XPos>Object.Value2
				Object.XPos+=Object.Value0
			else
				Object.Value0=0
				Object.XPos=Object.Value2
				if Object.Value4==1
					Object.State=5
				else
					Object.State=3
					Object.Frame=0
					Object.Value6=0
				end if
			end if
		end if
		break
	case 2
		Object.Value6++
		if Object.Value6==12
			Object.Value6=0
			Object.Frame++
		end if
		if Object.Frame>3
			Object.Frame=0
		end if
		if Object.Direction==FACING_LEFT
			Object.State=1
			Object.Frame=4
			Object.Value6=0
		end if
		break
	case 3
		Object.Value6++
		if Object.Value6==12
			Object.Value6=0
			Object.Frame++
		end if
		if Object.Frame>3
			Object.Frame=0
		end if
		if Object.Direction==FACING_RIGHT
			Object.State=1
			Object.Frame=4
			Object.Value6=0
		end if
		break
	case 4
		if Object.Value6<8
			Object.Frame=0
		else
			if Object.Value6>47
				if Object.Value5==0
					if Object.Value6<120
						Object.Value1=-262144
						Object.Value5=1
						Object.Frame=9
					else
						Object.Value6=0
					end if
				else
					if Object.Value1>0
						Object.Frame=0
					end if
				end if
			else
				temp0=Object.Value6
				temp0-=8
				temp0%=20
				temp0/=10
				Object.Frame=temp0
				Object.Frame+=9
			end if
		end if
		if Object.Value5==0
			if Object.Direction==FACING_LEFT
				Object.State=1
				Object.Frame=4
				Object.Value6=0
			end if
		end if
		Object.Value6++
		break
	case 5
		if Object.Value6<8
			Object.Frame=0
		else
			if Object.Value6>47
				if Object.Value5==0
					if Object.Value6<120
						Object.Value1=-262144
						Object.Value5=1
						Object.Frame=9
					else
						Object.Value6=0
					end if
				else
					if Object.Value1>0
						Object.Frame=0
					end if
				end if
			else
				temp0=Object.Value6
				temp0-=8
				temp0%=20
				temp0/=10
				Object.Frame=temp0
				Object.Frame+=9
			end if
		end if
		if Object.Value5==0
			if Object.Direction==FACING_RIGHT
				Object.State=1
				Object.Frame=4
				Object.Value6=0
			end if
		end if
		Object.Value6++
		break
	case 6
		Object.Value6++
		if Object.Value6==12
			Object.Value6=0
			Object.Frame++
		end if
		if Object.Frame>3
			Object.Frame=0
		end if

		BoxCollisionTest(C_TOUCH, object.entityPos, -4,-360,4,360, ArrayPos0, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO, HITBOX_AUTO)
		if CheckResult==0
			if object[ArrayPos0].Gravity==0
				Object.State=1
				Object.Frame=4
				Object.Value6=0
			end if
		else
			if object[ArrayPos0].Gravity==1
				Object.State=1
				Object.Frame=4
				Object.Value6=0
			end if
		end if
		break
	case 7
		object[ArrayPos0].Speed=0
		object[ArrayPos0].xvel=0
		object[ArrayPos0].Direction=Object.Direction
		if object[ArrayPos0].Animation==10
			Object.State=8
			Object.Frame=9
			Object.Value1=-262144
			Object.Value5=1
			if Object.Direction==FACING_RIGHT
				Object.Value0=-65536
			else
				Object.Value0=65536
			end if
		end if
		break
	case 8
		if Object.Value1>0
			Object.Frame=10
		end if
		if Object.Value0>0
			Object.Value0-=2048
		end if
		if Object.Value0<0
			Object.Value0+=2048
		end if
		Object.XPos+=Object.Value0
		if Object.Value5==0
			if Object.Value6==16
				Object.State=1
				Object.Frame=4
				Object.Value6=0
			else
				Object.Value6++
			end if
		end if
		if Object.XPos>Object.Value3
			Object.XPos=Object.Value3
		end if
		break

	end switch
	Object.Value7++
	if Object.State==7
		if Object.Value7>15
			CreateTempObject(TypeName[LoveHeart],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].iYPos-=6
			Object[TempObjectPos].InkEffect=2
			Object[TempObjectPos].Alpha=255
			Object.Value7=0
		end if
	else
		if Object.Value7>41
			CreateTempObject(TypeName[LoveHeart],0,Object.XPos,Object.YPos)
			Object[TempObjectPos].iYPos-=6
			Object[TempObjectPos].InkEffect=2
			Object[TempObjectPos].Alpha=255
			Object.Value7=0
		end if
	end if
	if Object.Value5==0
		ObjectTileGrip(0,0,19,0)
	else
		Object.Value1+=16384
		Object.YPos+=Object.Value1
		ObjectTileCollision(0,0,19,0)
		if CheckResult==1
			Object.Value1=0
			Object.Value5=0
		end if
	end if
	if Object.OutOfBounds==1
		Object.Priority=0
		Object.State=0
		Object.Value4=0
	end if
end event


event ObjectDraw
	DrawSpriteFX(Object.Frame,FX_FLIP,Object.XPos,Object.YPos)
end event


event ObjectStartup
	LoadSpriteSheet("R1/Objects.gif")
	ArrayPos0=32
	foreach (TypeName[AmyRose], ArrayPos0, ALL_ENTITIES)
		if Stage.PlayerListPos != PLAYER_SONIC // Amy should never appear if Sonic is not present
			Object[ArrayPos0].Type=TypeName[BlankObject]
		else
			if Options.GameMode==2
				Object[ArrayPos0].Type=TypeName[BlankObject]
			else
				Object[ArrayPos0].Value2=Object[ArrayPos0].XPos
				Object[ArrayPos0].Value2-=20971520
				Object[ArrayPos0].Value3=Object[ArrayPos0].XPos
				Object[ArrayPos0].Value3+=4194304
				Object[ArrayPos0].DrawOrder=4
			end if
		end if
	next
	SpriteFrame(-11,-20,22,40,34,208)
